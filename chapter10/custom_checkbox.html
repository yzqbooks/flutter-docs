                                                                                                                                                                                                                            # 10.6.1 CustomCheckbox

                                                                                                                                                                                                                            Flutter иҮӘеёҰзҡ„ Checkbox з»„д»¶жҳҜдёҚиғҪиҮӘз”ұжҢҮе®ҡеӨ§е°Ҹзҡ„пјҢжң¬иҠӮжҲ‘д»¬йҖҡиҝҮиҮӘе®ҡд№үдёҖдёӘеҸҜд»ҘиҮӘз”ұжҢҮе®ҡеӨ§е°Ҹзҡ„ CustomCheckbox з»„д»¶жқҘжј”зӨәеҰӮдҪ•йҖҡиҝҮе®ҡд№ү RenderObject зҡ„ж–№ејҸжқҘиҮӘе®ҡд№үз»„д»¶пјҲиҖҢдёҚжҳҜйҖҡиҝҮз»„еҗҲпјүгҖӮжҲ‘д»¬иҰҒе®һзҺ°зҡ„ CustomCheckbox з»„д»¶ж•ҲжһңеҰӮеӣҫ10-6жүҖзӨәпјҡ

                                                                                                                                                                                                                            1. жңүйҖүдёӯе’ҢжңӘйҖүдёӯдёӨз§ҚзҠ¶жҖҒгҖӮ
                                                                                                                                                                                                                            2. зҠ¶жҖҒеҲҮжҚўж—¶иҰҒжү§иЎҢеҠЁз”»гҖӮ
                                                                                                                                                                                                                            3. еҸҜд»ҘиҮӘе®ҡд№үеӨ–и§ӮгҖӮ

                                                                                                                                                                                                                            CustomCheckbox е®ҡд№үеҰӮдёӢпјҡ

                                                                                                                                                                                                                            import 'dart:math';
import 'package:flutter/material.dart';
import 'package:flutter/rendering.dart';
import 'package:flutter/scheduler.dart';

class CustomCheckbox extends LeafRenderObjectWidget {
  const CustomCheckbox({
    Key? key,
    this.strokeWidth = 2.0,
    this.value = false,
    this.strokeColor = Colors.white,
    this.fillColor = Colors.blue,
    this.radius = 2.0,
    this.onChanged,
  }) : super(key: key);

  final double strokeWidth; // вҖңеӢҫвҖқзҡ„зәҝжқЎе®ҪеәҰ
  final Color strokeColor; // вҖңеӢҫвҖқзҡ„зәҝжқЎе®ҪеәҰ
  final Color? fillColor; // еЎ«е……йўңиүІ
  final bool value; //йҖүдёӯзҠ¶жҖҒ
  final double radius; // еңҶи§’
  final ValueChanged<bool>? onChanged; // йҖүдёӯзҠ¶жҖҒеҸ‘з”ҹж”№еҸҳеҗҺзҡ„еӣһи°ғ

  @override
  RenderObject createRenderObject(BuildContext context) {
    return RenderCustomCheckbox(
      strokeWidth,
      strokeColor,
      fillColor ?? Theme.of(context).primaryColor,
      value,
      radius,
      onChanged,
    );
  }

  @override
  void updateRenderObject(context, RenderCustomCheckbox renderObject) {
    if (renderObject.value != value) {
      renderObject.animationStatus =
          value ? AnimationStatus.forward : AnimationStatus.reverse;
    }
    renderObject
      ..strokeWidth = strokeWidth
      ..strokeColor = strokeColor
      ..fillColor = fillColor ?? Theme.of(context).primaryColor
      ..radius = radius
      ..value = value
      ..onChanged = onChanged;
  }
}

  var rectProgress = Rect.lerp(
    rects[0],
    rects[1],
    // иғҢжҷҜеҠЁз”»зҡ„жү§иЎҢж—¶й•ҝжҳҜеүҚ 40% зҡ„ж—¶й—ҙ
    min(progress, bgAnimationInterval) / bgAnimationInterval,
  )!;
  final inner = RRect.fromRectXY(rectProgress, 0, 0);

  context.canvas.drawDRRect(outer, inner, paint);

                                                                                                                                                                                                                            # з»ҳеҲ¶еүҚжҷҜ

                                                                                                                                                                                                                            еүҚжҷҜжҳҜдёҖдёӘ"еӢҫ"пјҢе®ғжңүдёүдёӘзӮ№зҡ„иҝһзәҝжһ„жҲҗпјҢдёәдәҶз®ҖеҚ•иө·и§ҒпјҢжҲ‘д»¬е°Ҷиө·е§ӢзӮ№е’ҢдёӯзӮ№жӢҗзӮ№зҡ„дҪҚзҪ®ж №жҚ® Checkbox зҡ„еӨ§е°Ҹз®—еҮәеӣәе®ҡзҡ„еқҗж ҮпјҢз„¶еҗҺжҲ‘д»¬еңЁжҜҸдёҖеё§дёӯеҠЁжҖҒи°ғж•ҙз¬¬дёүдёӘзӮ№зҡ„дҪҚзҪ®е°ұеҸҜд»Ҙе®һзҺ°жү“еӢҫеҠЁз”»пјҡ

                                                                                                                                                                                                                            //з”» "еӢҫ"
void _drawCheckMark(PaintingContext context, Rect rect) {
  // еңЁз”»еҘҪиғҢжҷҜеҗҺеҶҚз”»еүҚжҷҜ
  if (progress > bgAnimationInterval) {
    
    //зЎ®е®ҡдёӯй—ҙжӢҗзӮ№дҪҚзҪ®
    final secondOffset = Offset(
      rect.left + rect.width / 2.5,
      rect.bottom - rect.height / 4,
    );
    // з¬¬дёүдёӘзӮ№зҡ„дҪҚзҪ®
    final lastOffset = Offset(
      rect.right - rect.width / 6,
      rect.top + rect.height / 4,
    );

    // жҲ‘д»¬еҸӘеҜ№з¬¬дёүдёӘзӮ№зҡ„дҪҚзҪ®еҒҡжҸ’еҖј
    final _lastOffset = Offset.lerp(
      secondOffset,
      lastOffset,
      (progress - bgAnimationInterval) / (1 - bgAnimationInterval),
    )!;

    // е°ҶдёүдёӘзӮ№иҝһиө·жқҘ
    final path = Path()
      ..moveTo(rect.left + rect.width / 7, rect.top + rect.height / 2)
      ..lineTo(secondOffset.dx, secondOffset.dy)
      ..lineTo(_lastOffset.dx, _lastOffset.dy);

    final paint = Paint()
      ..isAntiAlias = true
      ..style = PaintingStyle.stroke
      ..color = strokeColor
      ..strokeWidth = strokeWidth;

    context.canvas.drawPath(path, paint..style = PaintingStyle.stroke);
  }
}

                                                                                                                                                                                                                            # и°ғеәҰеҠЁз”»

                                                                                                                                                                                                                            жңҖеҗҺпјҢжҲ‘д»¬йңҖиҰҒи®©UIеҠЁиө·жқҘпјҢиҝҷж—¶жҲ‘д»¬еӣһжғідёҖдёӢеүҚйқўеҠЁз”»дёҖз« дёӯзҡ„еҶ…е®№пјҢдјҡж„ҸиҜҶеҲ° Flutter зҡ„еҠЁз”»жЎҶжһ¶жҳҜдҫқиө–дәҺ StatefulWidget зҡ„пјҢеҚіеҪ“зҠ¶жҖҒж”№еҸҳж—¶жҳҫејҸжҲ–йҡҗејҸзҡ„еҺ»и°ғз”Ё setState и§ҰеҸ‘жӣҙж–°гҖӮдҪҶжҳҜжҲ‘д»¬зӣҙжҺҘйҖҡиҝҮе®ҡд№ү RenderObject зҡ„ж–№ејҸжқҘе®һзҺ°зҡ„ CustomCheckboxпјҢе№¶дёҚжҳҜеҹәдәҺ StatefulWidget пјҢйӮЈиҜҘжҖҺд№ҲжқҘи°ғеәҰеҠЁз”»е‘ўпјҹжңүдёӨз§ҚеҠһжі•пјҡ

                                                                                                                                                                                                                            1. е°Ҷ CustomCheckbox з”ЁдёҖдёӘ StatefulWidget еҢ…иЈ…иө·жқҘпјҢиҝҷж ·е°ұеҸҜд»ҘеӨҚз”Ёд№ӢеүҚд»Ӣз»Қзҡ„жү§иЎҢеҠЁз”»зҡ„ж–№жі•гҖӮ
                                                                                                                                                                                                                            2. иҮӘе®ҡд№үеҠЁз”»и°ғеәҰгҖӮ

                                                                                                                                                                                                                            з¬¬дёҖз§Қж–№жі•зӣёдҝЎиҜ»иҖ…е·Із»ҸеҫҲзҶҹжӮүдәҶпјҢдёҚеҶҚиөҳиҝ°пјҢдёӢйқўжҲ‘д»¬жј”зӨәдёҖдёӢз¬¬дәҢз§Қж–№жі•пјҢжҲ‘д»¬зҡ„жҖқи·ҜжҳҜпјҡеңЁдёҖеё§з»ҳеҲ¶з»“жқҹеҗҺеҲӨж–ӯеҠЁз”»жҳҜеҗҰз»“жқҹпјҢеҰӮжһңеҠЁз”»жңӘз»“жқҹпјҢеҲҷе°Ҷе°ҶеҪ“еүҚз»„д»¶ж Үи®°дёәвҖқйңҖиҰҒйҮҚз»ҳвҖңпјҢз„¶еҗҺзӯүеҫ…дёӢдёҖеё§еҚіеҸҜпјҡ

                                                                                                                                                                                                                            void _scheduleAnimation() {
  if (_animationStatus != AnimationStatus.completed) {
    // йңҖиҰҒеңЁFlutter еҪ“еүҚframe з»“жқҹд№ӢеүҚеҶҚжү§иЎҢпјҢеӣ дёәдёҚиғҪеңЁз»ҳеҲ¶иҝҮзЁӢдёӯеҸҲе°Ҷз»„д»¶ж Үи®°дёәйңҖиҰҒйҮҚз»ҳ
    SchedulerBinding.instance!.addPostFrameCallback((Duration timeStamp) {
      if (_lastTimeStamp != null) {
        double delta = (timeStamp.inMilliseconds - _lastTimeStamp!) /
          duration.inMilliseconds;
        // еҰӮжһңжҳҜеҸҚеҗ‘еҠЁз”»пјҢеҲҷ progressеҖјиҰҒйҖҗжёҗеҮҸе°Ҹ
        if (_animationStatus == AnimationStatus.reverse) {
          delta = -delta;
        }
        //жӣҙж–°еҠЁз”»иҝӣеәҰ
        progress = progress + delta;
        
        if (progress >= 1 || progress <= 0) {
          //еҠЁз”»жү§иЎҢз»“жқҹ
          _animationStatus = AnimationStatus.completed;
          progress = progress.clamp(0, 1);
        }
      }
       //ж Үи®°дёәйңҖиҰҒйҮҚз»ҳ
      markNeedsPaint();
      _lastTimeStamp = timeStamp.inMilliseconds;
    });
  } else {
    _lastTimeStamp = null;
  }
}

                                                                                                                                                                                                                            # е“Қеә”зӮ№еҮ»дәӢд»¶

                                                                                                                                                                                                                            ж №жҚ®д№ӢеүҚдәӢд»¶еӨ„зҗҶзӣёе…із« иҠӮзҡ„д»Ӣз»ҚпјҢеҰӮжһңжҲ‘д»¬иҰҒи®©жёІжҹ“еҜ№иұЎиғҪеӨ„зҗҶдәӢд»¶пјҢеҲҷе®ғеҝ…йЎ»иғҪйҖҡиҝҮе‘ҪдёӯжөӢиҜ•пјҢд№ӢеҗҺжүҚиғҪеңЁ handleEvent ж–№жі•дёӯеӨ„зҗҶдәӢд»¶пјҢжүҖд»ҘжҲ‘д»¬йңҖиҰҒж·»еҠ еҰӮдёӢд»Јз Ғпјҡ

                                                                                                                                                                                                                            // еҝ…йЎ»зҪ®дёәtrueпјҢзЎ®дҝқиғҪйҖҡиҝҮе‘ҪдёӯжөӢиҜ•
@override
bool hitTestSelf(Offset position) => true;

// еҸӘжңүйҖҡиҝҮе‘ҪдёӯжөӢиҜ•пјҢжүҚдјҡи°ғз”Ёжң¬ж–№жі•пјҢжҲ‘д»¬еңЁжүӢжҢҮжҠ¬иө·ж—¶и§ҰеҸ‘дәӢд»¶еҚіеҸҜ
@override
void handleEvent(PointerEvent event, covariant BoxHitTestEntry entry) {
  if (event.down) {
    pointerId = event.pointer;
  } else if (pointerId == event.pointer) {
    // жүӢжҢҮжҠ¬иө·ж—¶и§ҰеҸ‘еӣһи°ғ
    onChanged?.call(!value);
  }
}

                                                                                                                                                                                                                            # 10.6.2 еҠЁз”»и°ғеәҰжҠҪиұЎ RenderObjectAnimationMixin

                                                                                                                                                                                                                            жҲ‘д»¬еҸҜд»ҘзңӢеҲ°пјҢеңЁRenderObject дёӯи°ғеәҰеҠЁз”»иҝҳжҳҜжҜ”иҫғеӨҚжқӮзҡ„пјҢдёәжӯӨжҲ‘д»¬жҠҪиұЎдәҶдёҖдёӘ RenderObjectAnimationMixinпјҢеҰӮжһңиҝҳжңүе…¶е®ғ RenderObject дёӯйңҖиҰҒжү§иЎҢеҠЁз”»пјҢеҸҜд»ҘзӣҙжҺҘеӨҚз”ЁгҖӮ

                                                                                                                                                                                                                            mixin RenderObjectAnimationMixin on RenderObject {
  double _progress = 0;
  int? _lastTimeStamp;

  // еҠЁз”»ж—¶й•ҝпјҢеӯҗзұ»еҸҜд»ҘйҮҚеҶҷ
  Duration get duration => const Duration(milliseconds: 200);
  AnimationStatus _animationStatus = AnimationStatus.completed;
  // и®ҫзҪ®еҠЁз”»зҠ¶жҖҒ
  set animationStatus(AnimationStatus v) {
    if (_animationStatus != v) {
      markNeedsPaint();
    }
    _animationStatus = v;
  }
  
  double get progress => _progress;
  set progress(double v) {
    _progress = v.clamp(0, 1);
  }

  @override
  void paint(PaintingContext context, Offset offset) {
    doPaint(context, offset); // и°ғз”Ёеӯҗзұ»з»ҳеҲ¶йҖ»иҫ‘
    _scheduleAnimation();
  }

  void _scheduleAnimation() {
    if (_animationStatus != AnimationStatus.completed) {
      SchedulerBinding.instance!.addPostFrameCallback((Duration timeStamp) {
        if (_lastTimeStamp != null) {
          double delta = (timeStamp.inMilliseconds - _lastTimeStamp!) /
              duration.inMilliseconds;
          
          //еңЁзү№е®ҡжғ…еҶөдёӢпјҢеҸҜиғҪеңЁдёҖеё§дёӯиҝһз»ӯзҡ„еҫҖframeCallbackдёӯж·»еҠ дәҶеӨҡж¬ЎпјҢеҜјиҮҙдёӨж¬Ўеӣһи°ғж—¶й—ҙй—ҙйҡ”дёә0пјҢ
          //иҝҷз§Қжғ…еҶөдёӢеә”иҜҘз»§з»ӯиҜ·жұӮйҮҚз»ҳгҖӮ
          if (delta == 0) {
            markNeedsPaint();
            return;
          }
          
          if (_animationStatus == AnimationStatus.reverse) {
            delta = -delta;
          }
          _progress = _progress + delta;
          if (_progress >= 1 || _progress <= 0) {
            _animationStatus = AnimationStatus.completed;
            _progress = _progress.clamp(0, 1);
          }
        }
        markNeedsPaint();
        _lastTimeStamp = timeStamp.inMilliseconds;
      });
    } else {
      _lastTimeStamp = null;
    }
  }

  // еӯҗзұ»е®һзҺ°з»ҳеҲ¶йҖ»иҫ‘зҡ„ең°ж–№
  void doPaint(PaintingContext context, Offset offset);
}

  final Color strokeColor; // вҖңеӢҫвҖқзҡ„зәҝжқЎе®ҪеәҰ

    );

    if (renderObject.value != value) {
      renderObject.animationStatus =
          value ? AnimationStatus.forward : AnimationStatus.reverse;

      ..strokeWidth = strokeWidth
      ..strokeColor = strokeColor
      ..fillColor = fillColor ?? Theme.of(context).primaryColor
      ..radius = radius
      ..value = value
      ..onChanged = onChanged;
  }
}

class RenderCustomCheckbox extends RenderBox with RenderObjectAnimationMixin {
  bool value;
  int pointerId = -1;
  double strokeWidth;
  Color strokeColor;
  Color fillColor;
  double radius;
  ValueChanged<bool>? onChanged;

  RenderCustomCheckbox(this.strokeWidth, this.strokeColor, this.fillColor,
      this.value, this.radius, this.onChanged) {
    progress = value ? 1 : 0;
  }

  @override
  bool get isRepaintBoundary => true;

  //иғҢжҷҜеҠЁз”»ж—¶й•ҝеҚ жҜ”пјҲиғҢжҷҜеҠЁз”»иҰҒеңЁеүҚ40%зҡ„ж—¶й—ҙеҶ…жү§иЎҢе®ҢжҜ•пјҢд№ӢеҗҺжү§иЎҢжү“еӢҫеҠЁз”»пјү
  final double bgAnimationInterval = .4;

  @override
  void doPaint(PaintingContext context, Offset offset) {
    Rect rect = offset & size;
    _drawBackground(context, rect);
    _drawCheckMark(context, rect);
  }

  void _drawBackground(PaintingContext context, Rect rect) {
    Color color = value ? fillColor : Colors.grey;
    var paint = Paint()
      ..isAntiAlias = true
      ..style = PaintingStyle.fill //еЎ«е……
      ..strokeWidth
      ..color = color;

    // жҲ‘д»¬еҜ№зҹ©еҪўеҒҡжҸ’еҖј
    final outer = RRect.fromRectXY(rect, radius, radius);
    var rects = [
      rect.inflate(-strokeWidth),
      Rect.fromCenter(center: rect.center, width: 0, height: 0)
    ];
    var rectProgress = Rect.lerp(
      rects[0],
      rects[1],
      min(progress, bgAnimationInterval) / bgAnimationInterval,
    )!;

    final inner = RRect.fromRectXY(rectProgress, 0, 0);
    // з”»иғҢжҷҜ
    context.canvas.drawDRRect(outer, inner, paint);
  }

  //з”» "еӢҫ"
  void _drawCheckMark(PaintingContext context, Rect rect) {
    // еңЁз”»еҘҪиғҢжҷҜеҗҺеҶҚз”»еүҚжҷҜ
    if (progress > bgAnimationInterval) {

      //зЎ®е®ҡдёӯй—ҙжӢҗзӮ№дҪҚзҪ®
      final secondOffset = Offset(
        rect.left + rect.width / 2.5,
        rect.bottom - rect.height / 4,
      );
      // з¬¬дёүдёӘзӮ№зҡ„дҪҚзҪ®
      final lastOffset = Offset(
        rect.right - rect.width / 6,
        rect.top + rect.height / 4,
      );

      // жҲ‘д»¬еҸӘеҜ№з¬¬дёүдёӘзӮ№зҡ„дҪҚзҪ®еҒҡжҸ’еҖј
      final _lastOffset = Offset.lerp(
        secondOffset,
        lastOffset,
        (progress - bgAnimationInterval) / (1 - bgAnimationInterval),
      )!;

      // е°ҶдёүдёӘзӮ№иҝһиө·жқҘ
      final path = Path()
        ..moveTo(rect.left + rect.width / 7, rect.top + rect.height / 2)
        ..lineTo(secondOffset.dx, secondOffset.dy)
        ..lineTo(_lastOffset.dx, _lastOffset.dy);

      final paint = Paint()
        ..isAntiAlias = true
        ..style = PaintingStyle.stroke
        ..color = strokeColor
        ..strokeWidth = strokeWidth;

      context.canvas.drawPath(path, paint..style = PaintingStyle.stroke);
    }
  }

  @override
  void performLayout() {
    // еҰӮжһңзҲ¶з»„д»¶жҢҮе®ҡдәҶеӣәе®ҡе®Ҫй«ҳпјҢеҲҷдҪҝз”ЁзҲ¶з»„д»¶жҢҮе®ҡзҡ„пјҢеҗҰеҲҷе®Ҫй«ҳй»ҳи®ӨзҪ®дёә 25
    size = constraints.constrain(
      constraints.isTight ? Size.infinite : const Size(25, 25),
    );
  }

  // еҝ…йЎ»зҪ®дёәtrueпјҢеҗҰеҲҷдёҚеҸҜд»Ҙе“Қеә”дәӢд»¶
  @override
  bool hitTestSelf(Offset position) => true;

  // еҸӘжңүйҖҡиҝҮзӮ№еҮ»жөӢиҜ•зҡ„з»„д»¶жүҚдјҡи°ғз”Ёжң¬ж–№жі•
  @override
  void handleEvent(PointerEvent event, covariant BoxHitTestEntry entry) {
    if (event.down) {
      pointerId = event.pointer;
    } else if (pointerId == event.pointer) {
      // еҲӨж–ӯжүӢжҢҮжҠ¬иө·ж—¶жҳҜеңЁз»„д»¶иҢғеӣҙеҶ…зҡ„иҜқжүҚи§ҰеҸ‘onChange
      if(size.contains(event.localPosition)) {
        onChanged?.call(!value);
      }
    }
  }
}
